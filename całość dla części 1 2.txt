import csv
import os

class BatteryLog:
    batt_id = 1 
    def __init__(self, id_baterii, temperatura_c):
      
        self.log_id = BatteryLog.batt_id 
        BatteryLog.batt_id += 1
        self.id_baterii = id_baterii
        self.temperatura_c = temperatura_c

        if self.temperatura_c > 60.0: 
            self.status = "PRZEGRZANA"
        else:
            self.status = "OK"

    def __str__(self): 
        return (f"LogID: {self.log_id}, "
                f"Bateria: {self.id_baterii}, "
                f"Temperatura: {self.temperatura_c}°C, "
                f"Status: {self.status}")


class LogManager:

    def __init__(self, nazwa_pliku="baterie_log.csv"):
        self.pomiary = []  
        self.nazwa_pliku = nazwa_pliku  

    def dodaj_pomiar(self, id_baterii, temperatura_c):
        log = BatteryLog(id_baterii, temperatura_c) 
        self.pomiary.append(log)

       
        self.zapisz_do_csv(log)
        return log

    def zapisz_do_csv(self, log):
        
        plik_istnieje = os.path.exists(self.nazwa_pliku)

        try:
            
            with open(self.nazwa_pliku, mode='a', newline='', encoding='utf-8') as plik_csv:
                writer = csv.writer(plik_csv, delimiter=';')

                if not plik_istnieje:
                    writer.writerow(["ID_LOGU", "ID_BATERII", "TEMPERATURA", "STATUS"])

                writer.writerow([log.log_id, log.id_baterii, log.temperatura_c, log.status])

        except PermissionError:
            print(
                f"\nPlik '{self.nazwa_pliku}' jest otwarty w innym programie.")
            print("Zamknij plik, aby program zapisywał logi!\n")

    def pobierz_historie_baterii(self, id_baterii):
        return [
            log for log in self.pomiary
            if log.id_baterii == id_baterii
        ]

    def pobierz_przegrzane(self):
        return [
            log for log in self.pomiary
            if log.status == "PRZEGRZANA"
        ]
