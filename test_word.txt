import os
import pandas as pd

class Lot:
    def __init__(self, numer_lotu, linia_lotnicza, kierunek, typ_operacji, planowana_godzina, opoznienie_min=0,
                 status="Planowany"):
        self.numer_lotu = numer_lotu
        self.linia_lotnicza = linia_lotnicza
        self.kierunek = kierunek
        self.typ_operacji = typ_operacji
        self.planowana_godzina = planowana_godzina
        self.opoznienie_min = opoznienie_min
        self.status = status

    def to_dict(self):
        return {
            "numer_lotu": self.numer_lotu,
            "linia_lotnicza": self.linia_lotnicza,
            "kierunek": self.kierunek,
            "typ_operacji": self.typ_operacji,
            "planowana_godzina": self.planowana_godzina,
            "opoznienie_min": self.opoznienie_min,
            "status": self.status
        }


class MenedzerLotniska:
    def __init__(self):
        self.loty = []

    def dodaj_lot(self, lot):
        self.loty.append(lot)

    def aktualizuj_opoznienie(self, numer_lotu, minuty):
        for lot in self.loty:
            if lot.numer_lotu == numer_lotu:
                lot.opoznienie_min = minuty
                if minuty > 0:
                    lot.status = "Opóźniony"
                else:
                    lot.status = "Planowany"
                return True
        return False

    def oblicz_srednie_opoznienie(self, linia_lotnicza):
        opoznienia = [lot.opoznienie_min for lot in self.loty if lot.linia_lotnicza == linia_lotnicza]
        return sum(opoznienia) / len(opoznienia) if opoznienia else 0

    def zapisz_rozklad(self, sciezka="rozklad.json"):
        dane_do_zapisu = [lot.to_dict() for lot in self.loty]
        with open(sciezka, 'w', encoding='utf-8') as plik:
            json.dump(dane_do_zapisu, plik, indent=4, ensure_ascii=False)

    def wczytaj_rozklad(self, sciezka="rozklad.json"):
        if not os.path.exists(sciezka):
            return
        with open(sciezka, 'r', encoding='utf-8') as plik:
            dane = json.load(plik)
            self.loty = []
            for e in dane:
                self.loty.append(Lot(e["numer_lotu"], e["linia_lotnicza"], e["kierunek"],
                                     e["typ_operacji"], e["planowana_godzina"],
                                     e.get("opoznienie_min", 0), e.get("status", "Planowany")))

    def generuj_raport_dobowy(self, sciezka="raport_dobowy.xlsx"):
        if not self.loty:
            return False
        dane_excel = [{
            "Numer Lotu": l.numer_lotu, "Linia Lotnicza": l.linia_lotnicza,
            "Kierunek": l.kierunek, "Typ": l.typ_operacji,
            "Planowany Czas": l.planowana_godzina, "Opóźnienie (min)": l.opoznienie_min,
            "Status": l.status
        } for l in self.loty]
        df = pd.DataFrame(dane_excel)
        df.to_excel(sciezka, index=False, engine='openpyxl')
        return True

# CZĘŚĆ 3: INTERFEJS GRAFICZNY I WIZUALIZACJA
# ==========================================
class AplikacjaLotniska(QMainWindow):
    def __init__(self):
        super().__init__()
        self.menedzer = MenedzerLotniska()
        self.menedzer.wczytaj_rozklad()

        self.setWindowTitle("System Kontroli i Rejestracji Ruchu Lotniczego")
        self.setGeometry(100, 100, 1200, 800)

        # Główny widget i układ
        glowny_widget = QWidget()
        self.setCentralWidget(glowny_widget)
        uklad_glowny = QVBoxLayout(glowny_widget)

        # --- GÓRNA SEKCJA: Formularze i Tabela ---
        uklad_gorny = QHBoxLayout()
        uklad_glowny.addLayout(uklad_gorny)

        # Panel boczny na formularze
        panel_boczny = QVBoxLayout()
        uklad_gorny.addLayout(panel_boczny, 1)

        # 1. Dodawanie nowego lotu
        panel_boczny.addWidget(QLabel("<b>DODAJ NOWY LOT:</b>"))
        self.input_numer = QLineEdit();
        self.input_numer.setPlaceholderText("Numer lotu (np. LO281)")
        panel_boczny.addWidget(self.input_numer)

        self.combo_linia = QComboBox()
        self.combo_linia.addItems(["PLL LOT", "Lufthansa", "Ryanair", "Wizz Air", "Air France"])
        panel_boczny.addWidget(self.combo_linia)

        self.input_kierunek = QLineEdit();
        self.input_kierunek.setPlaceholderText("Kierunek (Miasto)")
        panel_boczny.addWidget(self.input_kierunek)

        self.combo_typ = QComboBox()
        self.combo_typ.addItems(["Przylot", "Odlot"])
        panel_boczny.addWidget(self.combo_typ)

        self.input_czas = QLineEdit();
        self.input_czas.setPlaceholderText("Planowana godzina (np. 14:30)")
        panel_boczny.addWidget(self.input_czas)

        btn_dodaj = QPushButton("DODAJ LOT")
        btn_dodaj.clicked.connect(self.obsluga_dodaj_lot)
        panel_boczny.addWidget(btn_dodaj)

        panel_boczny.addSpacing(20)

        # 2. Aktualizacja opóźnienia
        panel_boczny.addWidget(QLabel("<b>AKTUALIZACJA OPÓŹNIENIA:</b>\n(Zaznacz lot w tabeli)"))
        self.spinbox_opoznienie = QSpinBox()
        self.spinbox_opoznienie.setRange(0, 1440)
        self.spinbox_opoznienie.setSuffix(" min")
        panel_boczny.addWidget(self.spinbox_opoznienie)

        btn_aktualizuj = QPushButton("AKTUALIZUJ STATUS")
        btn_aktualizuj.clicked.connect(self.obsluga_aktualizuj_status)
        panel_boczny.addWidget(btn_aktualizuj)

        panel_boczny.addStretch()

        # 3. Tabela Lotów (Smaczek: CSS Tablica Lotniskowa)
        self.tabela = QTableWidget()
        self.tabela.setColumnCount(7)
        self.tabela.setHorizontalHeaderLabels(
            ["Numer Lotu", "Linia Lotnicza", "Kierunek", "Typ", "Czas", "Opóźnienie", "Status"])
        self.tabela.horizontalHeader().setSectionResizeMode(QHeaderView.Stretch)
        self.tabela.setSelectionBehavior(QAbstractItemView.SelectRows)
        self.tabela.setSelectionMode(QAbstractItemView.SingleSelection)
        self.zastosuj_styl_tabeli()
        uklad_gorny.addWidget(self.tabela, 4)

        # --- DOLNA SEKCJA: Wykresy i Eksport ---
        uklad_dolny = QHBoxLayout()
        uklad_glowny.addLayout(uklad_dolny)

        # Panel do eksportu i wyboru wykresu
        panel_dolny_lewy = QVBoxLayout()
        uklad_dolny.addLayout(panel_dolny_lewy, 1)

        btn_eksport = QPushButton("EKSPORTUJ RAPORT DOBOWY (Excel)")
        btn_eksport.setStyleSheet("background-color: #2e7d32; color: white; font-weight: bold; padding: 10px;")
        btn_eksport.clicked.connect(self.obsluga_eksportu)
        panel_dolny_lewy.addWidget(btn_eksport)

        panel_dolny_lewy.addSpacing(20)
        panel_dolny_lewy.addWidget(QLabel("<b>Wybierz typ wizualizacji:</b>"))
        self.combo_wykres = QComboBox()
        self.combo_wykres.addItems(["Opcja 1: Udział linii w ruchu (Wykres kołowy)",
                                    "Opcja 2: Średnie opóźnienie linii (Wykres słupkowy)"])
        self.combo_wykres.currentIndexChanged.connect(self.odswiez_wykres)
        panel_dolny_lewy.addWidget(self.combo_wykres)
        panel_dolny_lewy.addStretch()

        # Matplotlib Canvas
        self.figure = Figure()
        self.canvas = FigureCanvas(self.figure)
        self.ax = self.figure.add_subplot(111)
        uklad_dolny.addWidget(self.canvas, 4)

        # Inicjalne załadowanie danych do GUI
        self.odswiez_tabele()

    def zastosuj_styl_tabeli(self):
        """Smaczek: Stylowanie CSS na profesjonalną czarną tablicę lotniskową"""
        self.tabela.setStyleSheet("""
            QTableWidget {
                background-color: #0d0d0d;
                color: #00ff00; /* Zielony tekst jak na terminalu */
                gridline-color: #333333;
                font-family: 'Courier New', Courier, monospace;
                font-size: 13px;
                font-weight: bold;
            }
            QHeaderView::section {
                background-color: #1a1a1a;
                color: #ffd700; /* Żółte nagłówki */
                font-weight: bold;
                border: 1px solid #333;
            }
            QTableWidget::item:selected {
                background-color: #333333;
                color: #ffffff;
            }
        """)

    def obsluga_dodaj_lot(self):
        numer = self.input_numer.text().strip()
        kierunek = self.input_kierunek.text().strip()
        czas = self.input_czas.text().strip()

        if not numer or not kierunek or not czas:
            QMessageBox.warning(self, "Błąd", "Wypełnij wszystkie pola tekstowe!")
            return

        nowy_lot = Lot(numer, self.combo_linia.currentText(), kierunek,
                       self.combo_typ.currentText(), czas)
        self.menedzer.dodaj_lot(nowy_lot)
        self.menedzer.zapisz_rozklad()

        # Czyszczenie formularza
        self.input_numer.clear()
        self.input_kierunek.clear()
        self.input_czas.clear()

        self.odswiez_ui()

    def obsluga_aktualizuj_status(self):
        wybrane_wiersze = self.tabela.selectionModel().selectedRows()
        if not wybrane_wiersze:
            QMessageBox.information(self, "Informacja", "Najpierw zaznacz lot w tabeli!")
            return

        wiersz = wybrane_wiersze[0].row()
        numer_lotu = self.tabela.item(wiersz, 0).text()
        minuty = self.spinbox_opoznienie.value()

        if self.menedzer.aktualizuj_opoznienie(numer_lotu, minuty):
            self.menedzer.zapisz_rozklad()
            self.odswiez_ui()
            self.spinbox_opoznienie.setValue(0)

    def obsluga_eksportu(self):
        sukces = self.menedzer.generuj_raport_dobowy()
        if sukces:
            QMessageBox.information(self, "Sukces", "Raport dobowy (Excel) został wygenerowany poprawnie.")
        else:
            QMessageBox.warning(self, "Błąd", "Brak danych do wygenerowania raportu.")

    def odswiez_ui(self):
        self.odswiez_tabele()
        self.odswiez_wykres()

    def odswiez_tabele(self):
        self.tabela.setRowCount(0)
        for row, lot in enumerate(self.menedzer.loty):
            self.tabela.insertRow(row)
            items = [
                QTableWidgetItem(lot.numer_lotu),
                QTableWidgetItem(lot.linia_lotnicza),
                QTableWidgetItem(lot.kierunek),
                QTableWidgetItem(lot.typ_operacji),
                QTableWidgetItem(lot.planowana_godzina),
                QTableWidgetItem(f"{lot.opoznienie_min} min"),
                QTableWidgetItem(lot.status)
            ]

            for col, item in enumerate(items):
                item.setFlags(item.flags() ^ Qt.ItemIsEditable)  # Blokada edycji z palca

                # Formatowanie warunkowe: jeśli opóźniony -> delikatnie czerwone tło
                if lot.status == "Opóźniony":
                    item.setBackground(QColor(60, 10, 10))  # Ciemno-czerwone tło
                    item.setForeground(QColor(255, 80, 80))  # Jasno-czerwony tekst

                self.tabela.setItem(row, col, item)

        self.odswiez_wykres()

    def odswiez_wykres(self):
        self.ax.clear()

        if not self.menedzer.loty:
            self.ax.text(0.5, 0.5, "Brak lotów w systemie", ha='center', va='center')
            self.canvas.draw()
            return

        indeks = self.combo_wykres.currentIndex()

        # OPCJA 1: Wykres Kołowy (Udział linii lotniczych)
        if indeks == 0:
            linie = [lot.linia_lotnicza for lot in self.menedzer.loty]
            zliczenia = {linia: linie.count(linia) for linia in set(linie)}

            self.ax.pie(zliczenia.values(), labels=zliczenia.keys(), autopct='%1.1f%%',
                        startangle=140, colors=['#4c72b0', '#dd8452', '#55a868', '#c44e52', '#8172b3'])
            self.ax.set_title("Udział linii lotniczych w całkowitym ruchu")

        # OPCJA 2: Wykres Słupkowy (Średnie opóźnienie w minutach)
        elif indeks == 1:
            unikalne_linie = list(set(lot.linia_lotnicza for lot in self.menedzer.loty))
            srednie_opoznienia = [self.menedzer.oblicz_srednie_opoznienie(linia) for linia in unikalne_linie]

            self.ax.bar(unikalne_linie, srednie_opoznienia, color='#c44e52')
            self.ax.set_title("Porównanie średniego opóźnienia linii lotniczych")
            self.ax.set_ylabel("Średnie opóźnienie [minuty]")
            self.ax.grid(axis='y', linestyle='--', alpha=0.7)

        self.canvas.draw()


if __name__ == "__main__":
    app = QApplication(sys.argv)

    # Ustawienie ogólnego stylu aplikacji, by pasował do mrocznej tablicy
    app.setStyle("Fusion")

    okno = AplikacjaLotniska()
    okno.show()
    sys.exit(app.exec_())
