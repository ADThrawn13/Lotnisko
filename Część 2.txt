    #To trzeba dodać do po funkcji def dodaj_pomiar i przed pobierz
    def zapisz_do_csv(self, log):
        # Sprawdzamy, czy plik już istnieje, aby wiedzieć, czy dodać nagłówki
        plik_istnieje = os.path.exists(self.nazwa_pliku)

        try:
            # Otwieranie pliku w trybie dopisywania ('a' - append)
            with open(self.nazwa_pliku, mode='a', newline='', encoding='utf-8') as plik_csv:
                # Używamy średnika jako separatora - Excel w polskiej wersji językowej lepiej to czyta
                writer = csv.writer(plik_csv, delimiter=';')

                # Jeśli plik nie istniał, najpierw zapisujemy nagłówki kolumn
                if not plik_istnieje:
                    writer.writerow(["ID_LOGU", "ID_BATERII", "TEMPERATURA", "STATUS"])

                # Zapisujemy nowy wiersz z danymi
                writer.writerow([log.log_id, log.id_baterii, log.temperatura_c, log.status])

        except PermissionError:
            # Przechwycenie błędu blokady zapisu (np. plik otwarty w Excelu)
            print(
                f"\nPlik '{self.nazwa_pliku}' jest prawdopodobnie otwarty w innym programie!!!")
