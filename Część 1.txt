class BatteryLog:

    batt_id = 1 #zmienna wspólna dla wszystkich class do log id, aby zaczyna³y siê od 1

    def __init__(self, id_baterii, temperatura_c):

        # zadeklarowanie zmiennych w classie, domyœlnie string
        self.log_id = BatteryLog.batt_id #nadawanie id_logu
        BatteryLog.batt_id += 1
        self.id_baterii = id_baterii
        self.temperatura_c = temperatura_c

        if self.temperatura_c > 60.0: #if czy temo przekracza
            self.status = "PRZEGRZANA"
        else:
            self.status = "OK"

    def __str__(self): #zwraca wyniki z funkcji, str wyœwietla siê po u¿yciu
        return (f"LogID: {self.log_id}, "
                f"Bateria: {self.id_baterii}, "
                f"Temperatura: {self.temperatura_c}°C, "
                f"Status: {self.status}")


class LogManager:

    def __init__(self):
        self.pomiary = [] #pusta lista, w której bêd¹ pomairy

    def dodaj_pomiar(self, id_baterii, temperatura_c):
        log = BatteryLog(id_baterii, temperatura_c) #zmienna dwuparametrowa
        self.pomiary.append(log)
        return log

    def pobierz_historie_baterii(self, id_baterii):
        return [
            log for log in self.pomiary
            if log.id_baterii == id_baterii
        ]

    def pobierz_przegrzane(self):
        return [
            log for log in self.pomiary
            if log.status == "PRZEGRZANA"
        ]


if __name__ == "__main__":
    manager = LogManager()
